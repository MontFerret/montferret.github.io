product-feed.fql
read-only workflow
// Load a JavaScript-rendered product page through CDP LET page = 'https://mockery.montferret.dev/scenarios/dynamic-products/basic/' LET doc = DOCUMENT(page, { driver: 'cdp' }) // Wait until product cards are available LET products = WAITFOR VALUE doc[~ css`.product-card`] WHEN LENGTH(.) > 0 TIMEOUT 5000 // Return clean, structured data FOR product IN products RETURN { brand: product[~? css`.product-brand`].textContent, title: product[~? css`.product-title`].textContent, price: TO_FLOAT(SUBSTITUTE(product[~? css`.product-price`], '$', '')) ON ERROR RETURN 0 }
Output shape [{ brand, title, price }]
