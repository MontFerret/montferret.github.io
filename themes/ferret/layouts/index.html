Ferret

Declarative data extraction

Extract with intent

FQL programs for structured extraction from websites, documents, APIs, and custom data sources.

Open playground Read the docs
browser.fql
read-only script
LET page = WEB::HTML::OPEN(@page, { driver: "cdp" }) LET products = WAITFOR VALUE page[~ `.product-card`] WHEN LENGTH(.) > 0 TIMEOUT 5000 ON TIMEOUT RETURN [] FOR product IN products RETURN { brand: product[~? `.product-brand`].textContent, title: product[~? `.product-title`].textContent, price: product[~? `.product-price`].textContent, }
api.fql
read-only script
LET api = NET::REST::CLIENT( "https://api.example.com") FOR order IN QUERY "/orders" IN api WITH { query: { status: "open", limit: 20 } } RETURN { id: order.id, customer: order.customer.name, total: order.total, status: order.status }
database.fql
read-only script
LET db = DB::SQLITE::OPEN({ path: "./data.db" }) FOR invoice IN QUERY ` SELECT id, customer, total FROM invoices WHERE total > ? ` IN db WITH { params: [@minTotal] } RETURN { id: invoice.id, customer: invoice.customer, total: invoice.total }
file.fql
read-only script
LET file = IO::FS::READ("./orders.csv") FOR row IN CSV::DECODE_ROWS(file) FILTER row.status == "failed" RETURN { orderId: row.order_id, customer: row.customer, reason: row.error }
