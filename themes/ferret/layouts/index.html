product-feed.fql
read-only script
LET doc = DOCUMENT(@page, { driver: "cdp" }) LET products = WAITFOR VALUE doc[~ css`.product-card`] WHEN LENGTH(.) > 0 TIMEOUT 5000 FOR product IN products RETURN { brand: product[~? css`.product-brand`].textContent, title: product[~? css`.product-title`].textContent, price: product[~? css`.product-price`].textContent, }
Output shape [{ brand, title, price }]
